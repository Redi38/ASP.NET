CREATE DATABASE textsettingsdb;

USE textsettingsdb;

CREATE TABLE TextStyles (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    BackgroundImage VARCHAR(255),
    BlockWidth VARCHAR(50),
    BlockMargin VARCHAR(50)
);
